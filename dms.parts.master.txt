-- PARTS MASTER

SELECT
t1.PART_NUM
,t1.PART_ID
,t1.PROD_COMPANY_CD
,t1.SWGL_SALES_CLASS_ID
,t1.SALES_CLASS_CD
,t1.PART_DESC_2
,t1.PART_DESC_1
,t1.PART_INTEGRATED_FLAG
,t1.FULF_STRATEGY_CD
,t1.PART_DATE_ENT
,t2.PROD_COMPANY_NAME
,t3.SALES_CLASS_DESC
,t3.COMMERCIAL_CATEGORY as Comm_Category
,t4.COST_PRICE/100 as PART_COST
,t5.TARGET_MKT_FCST_GROUP_DESC as Target_Mkt_Group
,t5.PRODUCT_GROUP_DESC as Product_Group
,t5.VALUE_STREAM_DESC

FROM 
dms.parts t1
,dms.product_companys t2
,dms.sdw_sales_classes t3
,dms.part_cost_calc_view t4
,dms.sdw_parts_rollup_view t5


WHERE
t1.PROD_COMPANY_CD = t2.PROD_COMPANY_CD(+)
AND t1.SWGL_SALES_CLASS_ID = t3.SALES_CLASS_ID (+)
AND t1.PART_ID = t4.PART_ID(+)
AND t1.PART_NUM = t5.PART_NUM(+)
